package project1;
import java.util.*;

class Student implements Comparable<Student>
{
	int rollno;
	String name;
	int age;
	Student(int rollno, String name, int age)
	{
		this.rollno=rollno;
		this.name=name;
		this.age=age;
	}
	public int compareTo(Student st)
	{
		if(age==st.age)
			return 0;
		else if(age>st.age)
			return 1;
		else
			return -1;
	}
}

public class TestCollection2 {

	public static void main(String[] args) 
	{
	ArrayList<Student> al=new ArrayList<Student>();
	al.add(new Student(101,"vish",21));

	Collections.sort(al);
	for(Student st:al)

	}

	}

}

{

	int age;
	Friends(int rollno, String name, int age)
	{
		this.rollno=rollno;
		this.name=name;
		this.age=age;
	}
	public int compareTo(Friends st)
	{
		if(age==st.age)
			return 0;
		else if(age<st.age)   //reverse order
			return 1;
		else
			return -1;
	}
}

package project1;
import java.util.*;
import java.io.*;

class Teacher
{
	int rollno;
	String name;
	int age;
	
	Teacher(int rollno, String name, int age)
	{
	this.rollno=rollno;
	this.name=name;
	this.age=age;
}
}

class AgeComparator implements Comparator<Teacher>
{
	public int compare(Teacher t1,Teacher t2)
	{
	if(t1.age==t2.age)
		return 0;
	else if(t1.age>t2.age)
		return 1;
	else
		return -1;
}
}
class NameComparator implements Comparator<Teacher>
{
	public int compare(Teacher t1,Teacher t2)
	{
		return t1.name.compareTo(t2.name);
	}
}

public class TestCollection4 {

	public static void main(String[] args) 
	{
		ArrayList<Teacher> al=new ArrayList<Teacher>();
		al.add(new Teacher(103,"ravi",30));
		al.add(new Teacher(106,"raju",29));
		al.add(new Teacher(108,"maruti",32));
		
		System.out.println("Sorting by Name");
		
		Collections.sort(al,new NameComparator());
		for(Teacher Teacher:al)
		{
			System.out.println(Teacher.rollno+" "+Teacher.name+" "+Teacher.age);
		}
		  
			System.out.println("Sorting by age");
			
			Collections.sort(al,new AgeComparator());
			for(Teacher Teacher:al)
			{
				System.out.println(Teacher.rollno+" "+Teacher.name+" "+Teacher.age);
			}

	}

}
